📖 Node.js readline Module Cheat Sheet

Read user input, process files line-by-line, build CLIs — Node.js 18+

💡 Golden Rule: Use readline for interactive user input (CLI prompts) or processing files line-by-line without loading entire files into memory. Import with node:readline prefix. For simple one-off questions, readline/promises is the modern way.
🖥️ Interactive CLI (readline/promises)
Interface Input

readline.createInterface

readline.createInterface({ input, output, completer?, terminal? })

Create a readline interface for reading from stdin (keyboard) and writing to stdout.

import readline from 'node:readline/promises';
import { stdin, stdout } from 'node:process';

// Create interface (Promise-based)
const rl = readline.createInterface({
  input: stdin,
  output: stdout
});

// Ask a question (returns Promise)
const answer = await rl.question('What is your name? ');
console.log(`Hello, ${answer}!`);

// Close the interface (important!)
rl.close();
✅ Use readline/promises for modern async/await usage. Always close the interface when done.
Input

rl.question (promises) / rl.on('line') (events)

rl.question(query, options?) / rl.on('line', handler) / rl.on('close', handler)

Get user input. Promise-based or event-based. line event fires on Enter key.

import readline from 'node:readline/promises';
import { stdin, stdout } from 'node:process';

const rl = readline.createInterface({ input: stdin, output: stdout });

// Promise-based (readline/promises)
try {
  const name = await rl.question('Your name? ');
  const age = await rl.question('Your age? ', { signal: AbortSignal.timeout(5000) });
  console.log(`${name} is ${age} years old`);
} catch (err) {
  if (err.name === 'AbortError') {
    console.log('Timed out!');
  }
} finally {
  rl.close();
}

// Event-based (traditional readline)
import readline from 'node:readline'; // Not promises
const rl2 = readline.createInterface({ input: stdin, output: stdout });
rl2.on('line', (input) => {
  console.log(`You typed: ${input}`);
  if (input === 'exit') rl2.close();
});
rl2.on('close', () => console.log('Goodbye!'));
✅ question() supports an optional signal option for timeouts/abort.
Interface

rl.write / rl.setPrompt / rl.prompt

rl.write(data) / rl.setPrompt(prompt) / rl.prompt()

Programmatically write to the readline interface, set a persistent prompt, and display it.

import readline from 'node:readline';
import { stdin, stdout } from 'node:process';

const rl = readline.createInterface({ input: stdin, output: stdout });

// Set a persistent prompt
rl.setPrompt('> ');
rl.prompt(); // Shows '> '

// Handle lines
rl.on('line', (line) => {
  if (line === 'exit') {
    rl.close();
  } else {
    console.log(`You said: ${line}`);
    rl.prompt(); // Show prompt again
  }
});

// Programmatically write to the interface
rl.write('Hello'); // Appears in input buffer
rl.write('\n'); // Submits the line
✅ Use setPrompt + prompt for persistent CLI prompts (like a REPL).
📄 Processing Files Line-by-Line
File

createInterface with file streams

readline.createInterface({ input: createReadStream, output? })

Process large files line-by-line without loading the whole file into memory.

import readline from 'node:readline';
import { createReadStream } from 'node:fs';

// Process a large file line by line
const fileStream = createReadStream('./huge-file.log');

const rl = readline.createInterface({
  input: fileStream,
  crlfDelay: Infinity // Handle both \r\n and \n
});

// Process each line
let lineCount = 0;
rl.on('line', (line) => {
  lineCount++;
  // Process line (e.g., parse JSON, filter, transform)
  if (line.includes('ERROR')) {
    console.log(`Error found: ${line}`);
  }
});

// File done
rl.on('close', () => {
  console.log(`Total lines: ${lineCount}`);
});

// Handle errors
rl.on('error', (err) => {
  console.error('Error reading file:', err);
});
✅ Great for processing log files, CSV files, or any large text file.
File

Reading file as async iterable

readline.createInterface returns AsyncIterable (Node.js 18+)

Use for await...of for cleaner line-by-line file processing.

import readline from 'node:readline';
import { createReadStream } from 'node:fs';

async function processFile(filePath) {
  const fileStream = createReadStream(filePath);
  const rl = readline.createInterface({
    input: fileStream,
    crlfDelay: Infinity
  });

  // Async iteration (Node.js 18+)
  for await (const line of rl) {
    // Process each line
    if (line.trim()) {
      console.log('Line:', line);
    }
  }
  
  console.log('File processing complete');
}

// Usage
await processFile('./data.txt');
✅ for await...of is the cleanest way to process files line-by-line in modern Node.js.
⚙️ Advanced Features
Util

Auto-completion (completer)

completer(line, callback) / completer(line) returns [[completions], matchedSubstring]

Add tab-completion to your CLI. Function receives the current input and returns suggestions.

import readline from 'node:readline';
import { stdin, stdout } from 'node:process';

const commands = ['help', 'exit', 'list', 'show', 'delete'];

const completer = (<line) => {
  const matches = commands.filter(cmd => cmd.startsWith(line));
  return [matches.length ? matches : commands, line];
};

const rl = readline.createInterface({
  input: stdin,
  output: stdout,
  completer
});

rl.on('line', (line) => {
  if (line === 'exit') rl.close();
  else if (line === 'help') console.log('Available commands: help, exit, list, show, delete');
  else console.log(`Unknown command: ${line}`);
  rl.prompt();
});

rl.setPrompt('> ');
rl.prompt();
✅ completer can return async results (callback) or sync results (array).
Util

rl.line / rl.cursor / rl.getCursorPos

rl.line (current line), rl.cursor (cursor position), rl.getCursorPos()

Access the current input line and cursor position. Useful for advanced CLI interactions.

import readline from 'node:readline';
import { stdin, stdout } from 'node:process';

const rl = readline.createInterface({
  input: stdin,
  output: stdout,
  terminal: true
});

// Listen for keypress events
rl.input.on('keypress', (str, key) => {
  if (key.name === 'tab') {
    console.log(`\nCurrent line: "${rl.line}"`);
    console.log(`Cursor at: ${rl.cursor}`);
    const pos = rl.getCursorPos();
    console.log(`Cursor position: row ${pos.rows}, col ${pos.cols}`);
  }
});

// Enable raw mode for keypress events
rl.input.setRawMode(true);
rl.input.resume();

rl.on('close', () => process.exit(0));
⚠️ Requires terminal: true. Use with care — advanced feature.
Util

rl.clearLine / rl.clearScreenDown / rl.moveCursor

rl.clearLine(dir) / rl.clearScreenDown() / rl.moveCursor(dx, dy)

Manipulate the terminal cursor and clear lines/screens. Great for animated CLIs.

import readline from 'node:readline';
import { stdin, stdout } from 'node:process';

const rl = readline.createInterface({
  input: stdin,
  output: stdout,
  terminal: true
});

// Animation: loading spinner
const spinner = ['⠋', '⠙', '⠹', '⠸', '⠼', '⠴', '⠦', '⠧', '⠇', '⠏'];
let i = 0;
const interval = setInterval(() => {
  // Clear the current line
  rl.clearLine(0); // 0 = cursor to start
  rl.cursorTo(0); // Move cursor to beginning
  rl.write(`Loading ${spinner[i % spinner.length]}`);
  i++;
}, 100);

// Clear the screen down from cursor
setTimeout(() => {
  clearInterval(interval);
  rl.clearLine(0);
  rl.cursorTo(0);
  rl.write('✅ Done!');
  rl.close();
}, 5000);
✅ Use for progress bars, loading animations, and interactive terminal UIs.
🧪 Legacy readline (Callback-based)
Interface

Traditional readline (callback)

rl.question(query, callback) / rl.on('line', callback)

The original callback-based API. Still works but readline/promises is preferred.

import readline from 'node:readline';
import { stdin, stdout } from 'node:process';

const rl = readline.createInterface({ input: stdin, output: stdout });

// Callback-based question
rl.question('What is your name? ', (name) => {
  rl.question('What is your age? ', (age) => {
    console.log(`${name} is ${age} years old`);
    rl.close();
  });
});

// Event-based
const rl2 = readline.createInterface({ input: stdin, output: stdout });
rl2.on('line', (line) => {
  if (line === 'exit') rl2.close();
  else console.log(`Echo: ${line}`);
});
⚠️ Legacy API — use readline/promises for new code with async/await.
🚀 Pro Pattern — Robust CLI with Validation: 
import readline from 'node:readline/promises';
import { stdin, stdout } from 'node:process';

const rl = readline.createInterface({ input: stdin, output: stdout });

async function prompt(query, validator, errorMsg = 'Invalid input') {
  while (true) {
    const answer = await rl.question(query);
    const result = await validator(answer);
    if (result.valid) return result.value;
    console.log(errorMsg);
  }
}

// Usage with validation
const name = await prompt(
  'Enter your name: ',
  input => ({
    valid: input.length > 0,
    value: input.trim()
  }),
  'Name cannot be empty'
);

const age = await prompt(
  'Enter your age: ',
  input => {
    const num = Number(input);
    return {
      valid: !isNaN(num) && num > 0,
      value: num
    };
  },
  'Please enter a valid number'
);

console.log(`Name: ${name}, Age: ${age}`);
rl.close();
