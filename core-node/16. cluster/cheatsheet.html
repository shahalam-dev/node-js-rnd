👥 Node.js cluster Module Cheat Sheet

Multi-process architecture for performance and reliability — Node.js 18+

💡 Golden Rule: Use cluster to leverage multi-core CPUs. The master process manages worker processes. Workers share server ports via the master. Use node:cluster prefix. Important: Clusters are for Node.js processes only, not for arbitrary child processes.
👑 Master & Worker Setup
Master Worker

cluster.isMaster / cluster.isWorker

cluster.isMaster / cluster.isWorker (Boolean)

Check if the current process is the master or a worker. Use to define different behavior.

import cluster from 'node:cluster';
import os from 'node:os';
import http from 'node:http';

if (cluster.isMaster) {
  // Master process: fork workers
  const numCPUs = os.availableParallelism();
  console.log(`Master ${process.pid} forking ${numCPUs} workers`);

  for (let i = 0; i < numCPUs; i++) {
    cluster.fork();
  }
} else {
  // Worker process: create server
  const server = http.createServer((req, res) => {
    res.writeHead(200);
    res.end(`Hello from worker ${process.pid}`);
  });

  server.listen(3000, () => {
    console.log(`Worker ${process.pid} listening`);
  });
}
✅ isMaster and isWorker are the primary way to separate master and worker logic.
Master

cluster.fork / cluster.setupMaster

cluster.fork(env?) / cluster.setupMaster(settings?)

Fork a new worker process. setupMaster configures how workers are forked (e.g., different script, args).

import cluster from 'node:cluster';

if (cluster.isMaster) {
  // Setup master configuration
  cluster.setupMaster({
    exec: 'worker.js',           // Custom worker file
    args: ['--harmony'],          // Args passed to workers
    silent: false                // Inherit stdout/stderr
  });

  // Fork with environment variables
  const worker = cluster.fork({
    CUSTOM_VAR: 'value',
    NODE_OPTIONS: '--max-old-space-size=1024'
  });

  // Fork multiple workers
  for (let i = 0; i < 4; i++) {
    cluster.fork();
  }
}
⚠️ setupMaster must be called before any workers are forked.
💬 IPC Communication
IPC

worker.send / cluster.on('message')

worker.send(message, handle?, callback?) / cluster.on('message', (worker, message, handle) => {})

Send messages between master and workers via IPC. Messages are serialized as JSON.

import cluster from 'node:cluster';

if (cluster.isMaster) {
  // Fork a worker
  const worker = cluster.fork();

  // Send message to worker
  worker.send({ type: 'greeting', data: 'Hello worker' });

  // Receive messages from workers
  cluster.on('message', (worker, message, handle) => {
    console.log(`Master received from ${worker.process.pid}:`, message);
    
    // Can send handle (e.g., socket) for advanced use
    if (handle) {
      // handle is a net.Socket or similar
    }
  });
} else {
  // Worker: receive messages from master
  process.on('message', (message) => {
    console.log(`Worker ${process.pid} received:`, message);
    
    // Reply to master
    process.send({ type: 'ack', pid: process.pid });
  });
}
✅ IPC is JSON-serialized. For large data, use shared memory or files.
IPC

cluster.on('online') / cluster.on('listening')

cluster.on('online', worker => {}) / cluster.on('listening', (worker, address) => {})

Detect when workers come online and when they start listening on a server.

import cluster from 'node:cluster';

if (cluster.isMaster) {
  const workers = [];

  // When a worker is online
  cluster.on('online', (worker) => {
    console.log(`Worker ${worker.process.pid} is online`);
  });

  // When a worker starts listening
  cluster.on('listening', (worker, address) => {
    console.log(
      `Worker ${worker.process.pid} listening on ${address.address}:${address.port}`
    );
    workers.push(worker);
  });

  // Fork workers
  for (let i = 0; i < 4; i++) {
    cluster.fork();
  }
}
✅ Use 'listening' to know when workers are ready to accept connections.
🔄 Worker Management
Worker

cluster.workers / cluster.worker

cluster.workers (Object) / cluster.worker (Worker object)

cluster.workers holds all worker objects in the master. cluster.worker gives the current worker its own object.

import cluster from 'node:cluster';

if (cluster.isMaster) {
  // Fork some workers
  for (let i = 0; i < 4; i++) {
    cluster.fork();
  }

  // Access all workers
  console.log(`Total workers: ${Object.keys(cluster.workers).length}`);

  // Iterate over workers
  for (const id in cluster.workers) {
    const worker = cluster.workers[id];
    console.log(`Worker ${worker.process.pid}`);
  }
} else {
  // Worker: access its own info
  console.log(`Worker ID: ${cluster.worker.id}, PID: ${process.pid}`);
  
  // Worker can get its own properties
  console.log(`Worker is connected: ${cluster.worker.isConnected()}`);
}
✅ cluster.workers is only available in the master process. cluster.worker is only available in workers.
Worker

worker.kill / worker.disconnect

worker.kill(signal?) / worker.disconnect()

kill terminates the worker process. disconnect gracefully disconnects the IPC channel.

import cluster from 'node:cluster';

if (cluster.isMaster) {
  const worker = cluster.fork();

  // Graceful disconnect (allows worker to finish tasks)
  worker.disconnect();
  worker.on('disconnect', () => {
    console.log(`Worker ${worker.process.pid} disconnected`);
    // Force kill after disconnect
    worker.kill('SIGTERM');
  });

  // Force kill immediately
  worker.kill('SIGKILL'); // Immediate termination

  // Listen for death events
  cluster.on('exit', (worker, code, signal) => {
    console.log(`Worker ${worker.process.pid} died (${code || signal})`);
  });
}
✅ Use disconnect for graceful shutdown. Use kill for immediate termination.
Worker

cluster.on('exit') & Auto-restart

cluster.on('exit', (worker, code, signal) => {})

Handle worker exits and implement auto-restart for reliability.

import cluster from 'node:cluster';
import os from 'node:os';

if (cluster.isMaster) {
  const numCPUs = os.availableParallelism();

  // Fork initial workers
  for (let i = 0; i < numCPUs; i++) {
    cluster.fork();
  }

  // Auto-restart on exit
  cluster.on('exit', (worker, code, signal) => {
    console.log(`Worker ${worker.process.pid} died. Restarting...`);
    
    // Fork a new worker to replace the dead one
    const newWorker = cluster.fork();
    
    // Optional: add delay before restart to avoid thrashing
    // setTimeout(() => cluster.fork(), 1000);
  });

  // Handle worker crashes with error codes
  cluster.on('exit', (worker, code) => {
    if (code !== 0 && !worker.exitedAfterDisconnect) {
      console.log(`Worker crashed (code: ${code}). Restarting...`);
    }
  });
}
⚠️ Check worker.exitedAfterDisconnect to distinguish between intentional and accidental exits.
🔧 Advanced Patterns
Util

cluster.schedulingPolicy

cluster.schedulingPolicy (cluster.SCHED_RR / cluster.SCHED_NONE)

Control how the master distributes connections among workers. Round-robin (default) or OS scheduling.

import cluster from 'node:cluster';

// Round-robin scheduling (default, recommended)
cluster.schedulingPolicy = cluster.SCHED_RR;

// OS scheduling (may distribute unevenly)
cluster.schedulingPolicy = cluster.SCHED_NONE;

// Check current policy
console.log('Scheduling policy:', cluster.schedulingPolicy);

// Set before any workers are forked
if (cluster.isMaster) {
  cluster.schedulingPolicy = cluster.SCHED_RR;
  for (let i = 0; i < 4; i++) {
    cluster.fork();
  }
}
✅ Round-robin is usually the best choice for HTTP servers. OS scheduling can cause uneven load distribution.
Util

Worker Object Properties

worker.process.pid, worker.id, worker.isDead(), worker.isConnected()

Various properties and methods on the Worker object.

import cluster from 'node:cluster';

if (cluster.isMaster) {
  const worker = cluster.fork();

  // Worker properties
  console.log('Worker ID:', worker.id);
  console.log('Worker PID:', worker.process.pid);
  console.log('Worker connected:', worker.isConnected());
  console.log('Worker dead:', worker.isDead());

  // Worker events
  worker.on('online', () => console.log('Worker online'));
  worker.on('listening', () => console.log('Worker listening'));
  worker.on('disconnect', () => console.log('Worker disconnected'));
  worker.on('exit', (code, signal) => {
    console.log(`Worker exited: ${code || signal}`);
  });
}
Util

Shared Server Example

Server sharing with net/http

Workers automatically share server ports via the master. No special code needed.

import cluster from 'node:cluster';
import http from 'node:http';
import os from 'node:os';

if (cluster.isMaster) {
  // Master: just fork workers
  for (let i = 0; i < os.availableParallelism(); i++) {
    cluster.fork();
  }
} else {
  // Worker: create server as usual
  const server = http.createServer(req => {
    // Each worker handles requests independently
    console.log(`Handled by ${process.pid}`);
  });

  // All workers listen on the same port
  // The master automatically distributes connections
  server.listen(3000, () => {
    console.log(`Worker ${process.pid} ready`);
  });
}
✅ The master handles incoming connections and distributes them to workers. Workers don't need to coordinate.
🚀 Pro Pattern — Production-Ready Cluster: 
import cluster from 'node:cluster';
import os from 'node:os';
import http from 'node:http';

if (cluster.isMaster) {
  // Set up workers = CPU cores, but leave one for OS
  const numWorkers = Math.max(1, os.availableParallelism() - 1);
  
  // Fork workers
  for (let i = 0; i < numWorkers; i++) {
    cluster.fork();
  }

  // Auto-restart on crash with exponential backoff
  let restartDelay = 1000;
  cluster.on('exit', (worker, code) => {
    if (!worker.exitedAfterDisconnect) {
      console.log(`Worker crashed. Restarting in ${restartDelay}ms`);
      setTimeout(() => {
        cluster.fork();
        restartDelay = Math.min(restartDelay * 1.5, 30000);
      }, restartDelay);
    }
  });

  // Graceful shutdown
  const gracefulShutdown = () => {
    console.log('Shutting down cluster...');
    for (const id in cluster.workers) {
      const worker = cluster.workers[id];
      worker.disconnect();
    }
    setTimeout(() => process.exit(0), 5000).unref();
  };

  process.on('SIGTERM', gracefulShutdown);
  process.on('SIGINT', gracefulShutdown);
} else {
  // Worker: create app
  const app = http.createServer((req, res) => {
    res.end(`Worker ${process.pid}\n`);
  });
  
  const server = app.listen(3000);
  
  // Clean up on disconnect
  process.on('disconnect', () => {
    server.close(() => {
      process.exit(0);
    });
  });
}

⚠️ Common Pitfalls

MistakeWhy It's WrongFix
Forking more workers than CPU coresWastes resources, context switchingUse os.availableParallelism()
Not handling exit eventsWorkers die silently, reduced capacityAlways restart crashed workers
Stateful data in workersWorkers don't share memoryUse external stores (Redis, DB)
Using cluster.workers in workerReturns {} in workersOnly use in master
Not checking exitedAfterDisconnectRestarts workers that were intentionally killedCheck before restarting
Using cluster.fork() inside workersWorkers can't fork other workersOnly master can fork
Assuming all workers are identicalEnvironment can differPass configuration via env/args
