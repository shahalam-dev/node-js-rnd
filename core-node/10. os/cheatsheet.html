🖥️ Node.js os Module Cheat Sheet

Operating system utilities — Node.js 18+

💡 Golden Rule: The os module provides platform-independent system information. Always use os methods instead of platform-specific commands or parsing process output. Import with node:os prefix.
💻 System Information
System

os.hostname / os.type / os.platform

os.hostname() / os.type() / os.platform()

Get system hostname, OS name, and platform identifier.

import os from 'node:os';

os.hostname();  // 'my-server'
os.type();       // 'Linux' | 'Darwin' | 'Windows_NT'
os.platform();   // 'linux' | 'darwin' | 'win32'

// Detect OS
if (os.platform() === 'win32') {
  console.log('Running on Windows');
} else if (os.platform() === 'darwin') {
  console.log('Running on macOS');
} else {
  console.log('Running on Linux/Unix');
}
✅ Use os.platform() for conditional logic based on OS.
System

os.release / os.version / os.uptime

os.release() / os.version() / os.uptime()

Get OS version, kernel release, and system uptime in seconds.

import os from 'node:os';

os.release();   // '5.15.0-91-generic' (Linux kernel)
os.version();   // '#73-Ubuntu SMP...' or 'Windows 10 Pro'
os.uptime();    // 1234567 (seconds since boot)

// Human-readable uptime
function formatUptime(seconds) {
  const days = Math.floor(seconds / 86400);
  const hours = Math.floor((seconds % 86400) / 3600);
  return `${days}d ${hours}h`;
}
formatUptime(os.uptime());
ℹ️ os.uptime() returns seconds. Divide by 60 for minutes, 3600 for hours.
System

os.endianness / os.arch / os.machine

os.endianness() / os.arch() / os.machine()

Get CPU architecture, endianness, and machine type.

import os from 'node:os';

os.endianness();  // 'LE' (Little Endian) or 'BE' (Big Endian)
os.arch();        // 'x64' | 'arm64' | 'ia32' | 'ppc' | etc.
os.machine();     // 'x86_64' | 'aarch64' (more detailed)

// Check architecture
if (os.arch() === 'arm64') {
  console.log('Running on Apple Silicon / ARM64');
}
✅ Use os.arch() for conditional code paths based on architecture.
🧠 Memory & CPU
Memory

os.totalmem / os.freemem

os.totalmem() / os.freemem()

Get total and free system memory in bytes.

import os from 'node:os';

const total = os.totalmem();  // bytes
const free = os.freemem();    // bytes
const used = total - free;

// Convert to GB
const totalGB = (total / 1024 ** 3).toFixed(2);
const freeGB = (free / 1024 ** 3).toFixed(2);

console.log(`Total: ${totalGB} GB, Free: ${freeGB} GB`);
// → Total: 16.00 GB, Free: 8.42 GB
⚠️ Values are in bytes. Divide by 1024 ** 3 for GB, 1024 ** 2 for MB.
CPU

os.cpus

os.cpus()

Get information about each CPU/core: model, speed, and times (idle, user, sys).

import os from 'node:os';

const cpus = os.cpus();
cpus.length;  // Number of CPU cores

// Each CPU object:
cpus[0] // → {
  //   model: 'Intel(R) Core(TM) i7-8700K',
  //   speed: 3700, // MHz
  //   times: { user: 123456, nice: 0, sys: 78901, idle: 987654, irq: 0 }
  // }

// Get total CPU time
const totalTime = cpus.reduce((sum, cpu) => {
  return sum + cpu.times.user + cpu.times.sys + cpu.times.idle;
}, 0);

// Calculate CPU usage percentage
function getCPUUsage() {
  const start = getCPUInfo();
  const startIdle = start.idle;
  const startTotal = start.total;
  
  // Wait a bit (or use setInterval)
  const end = getCPUInfo();
  const idleDiff = end.idle - startIdle;
  const totalDiff = end.total - startTotal;
  
  return 100 - (idleDiff / totalDiff) * 100;
}
✅ Use os.cpus().length to get number of CPU cores for parallelism.
CPU

os.loadavg

os.loadavg()

Get system load averages for 1, 5, and 15 minutes. (UNIX-only; on Windows returns [0,0,0])

import os from 'node:os';

const [oneMin, fiveMin, fifteenMin] = os.loadavg();
// → [1.23, 1.45, 1.67]

const cores = os.cpus().length;
const usage = (oneMin / cores) * 100;
console.log(`CPU usage: ${usage.toFixed(2)}%`);

// Check if system is overloaded
if (oneMin > cores * 0.8) {
  console.warn('System is under high load!');
}
⚠️ Load average is UNIX-specific. On Windows, it always returns [0,0,0].
🌐 Network
Network

os.networkInterfaces

os.networkInterfaces()

Get network interfaces with IP addresses, MAC, family, and more.

import os from 'node:os';

const interfaces = os.networkInterfaces();
// → {
//   eth0: [
//     { address: '192.168.1.100', netmask: '255.255.255.0', family: 'IPv4', mac: '00:11:22:33:44:55', internal: false, cidr: '192.168.1.100/24' },
//     { address: 'fe80::211:22ff:fe33:4455', netmask: 'ffff:ffff:ffff:ffff::', family: 'IPv6', mac: '00:11:22:33:44:55', internal: false, cidr: 'fe80::211:22ff:fe33:4455/64' }
//   ],
//   lo0: [ ... ]
// }

// Get local IP addresses (non-internal)
const ips = Object.values(interfaces)
  .flat()
  .filter(iface => !iface.internal && iface.family === 'IPv4')
  .map(iface => iface.address);

// Get first non-internal IPv4 address
const localIP = ips[0] || '127.0.0.1';
✅ Use for: getting server IP addresses, network diagnostics.
👤 User & System Paths
User

os.userInfo / os.homedir / os.tmpdir

os.userInfo() / os.homedir() / os.tmpdir()

Get current user information, home directory, and temp directory path.

import os from 'node:os';

// User info
const user = os.userInfo();
// → {
//   username: 'john',
//   uid: 1000,
//   gid: 1000,
//   shell: '/bin/bash',
//   homedir: '/home/john',
//   dir: '/home/john'
// }

os.homedir();  // '/home/john' or 'C:\Users\john'
os.tmpdir();   // '/tmp' or 'C:\Users\john\AppData\Local\Temp'

// Safe temp file path
import path from 'node:path';
const tempFile = path.join(os.tmpdir(), 'my-temp-file.txt');
✅ Always use os.tmpdir() for temporary files (cross-platform).
Constants

os.EOL / os.constants

os.EOL / os.constants

End-of-line character (\n on POSIX, \r\n on Windows) and error signal constants.

import os from 'node:os';

// End-of-line character
os.EOL;  // '\n' or '\r\n' (platform-specific)

// Write OS-specific newlines
fs.writeFileSync('log.txt', `Line 1${os.EOL}Line 2`);

// Constants (error codes, signals, etc.)
os.constants.signals.SIGKILL;  // 'SIGKILL'
os.constants.errno.EACCES;  // 'EACCES' (permission denied)

// Check if a specific constant exists
if (os.constants.errno.ENOENT) {
  // ENOENT exists on this platform
}
✅ Use os.EOL for cross-platform newline handling.
Util

os.setPriority / os.getPriority

os.getPriority(pid?) / os.setPriority(pid?, priority)

Get or set process scheduling priority. Higher priority = more CPU time.

import os from 'node:os';

// Get current process priority
const current = os.getPriority();  // -20 to 19 (higher = more priority)

// Set priority for current process
os.setPriority(10);  // lower priority (nice)

// Set priority for another process
os.setPriority(12345, 5);  // pid, priority

// Constants for priority
const { PRIORITY_LOW, PRIORITY_HIGH } = os.constants.priority;
os.setPriority(PRIORITY_LOW);
⚠️ Setting priority may require elevated permissions. Not available on all platforms.
🚀 Pro Pattern — System Health Monitor: 
import os from 'node:os';

function getSystemHealth() {
  const totalMem = os.totalmem();
  const freeMem = os.freemem();
  const usedMem = totalMem - freeMem;
  const memPercent = (usedMem / totalMem) * 100;

  const cores = os.cpus().length;
  const [load1, load5, load15] = os.loadavg();
  const cpuPercent = (load1 / cores) * 100;

  return {
    platform: os.platform(),
    hostname: os.hostname(),
    uptime: os.uptime(),
    memory: {
      total: (totalMem / 1024 ** 3).toFixed(2) + ' GB',
      used: (usedMem / 1024 ** 3).toFixed(2) + ' GB',
      percent: memPercent.toFixed(1) + '%'
    },
    cpu: {
      cores: cores,
      load1: load1.toFixed(2),
      load5: load5.toFixed(2),
      percent: cpuPercent.toFixed(1) + '%'
    }
  };
}

// Log health every 30 seconds
setInterval(() => {
  console.log(getSystemHealth());
}, 30000);

MistakeWhy It's WrongFix
os.platform() === 'windows'Platform returns 'win32', not 'windows'os.platform() === 'win32'
os.loadavg() on WindowsAlways returns [0,0,0] on WindowsCheck os.platform() before using
Hardcoding '/tmp'Breaks on WindowsUse os.tmpdir()
Hardcoding \nWindows uses \r\nUse os.EOL
Memory values in bytesLarge numbers are hard to readConvert to MB or GB with / 1024 ** 2 or / 1024 ** 3
