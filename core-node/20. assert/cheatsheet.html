✅ Node.js assert Module Cheat Sheet

Testing assertions and value validation — Node.js 18+

💡 Golden Rule: The assert module provides a set of assertion functions for testing invariants. Use strict mode (assert.strict) for reliable comparisons without coercion. Import with node:assert prefix. For production, use assert for validation; for tests, use a full testing framework.
⚖️ Equality Assertions
Equality

assert.equal / strictEqual

assert.equal(actual, expected, message?) / assert.strictEqual(actual, expected, message?)

equal uses loose equality (==). strictEqual uses strict equality (===). Prefer strictEqual.

import assert from 'node:assert';
import { strict } from 'node:assert';

// Loose equality (==) - avoid!
assert.equal(1, '1'); // Passes

// Strict equality (===) - recommended
assert.strictEqual(1, 1); // Passes
assert.strictEqual(1, '1'); // Throws

// Using strict import (all methods are strict by default)
strict.equal(1, '1'); // Throws (strict equal)
strict.equal(1, 1);  // Passes

// With custom message
assert.strictEqual(1, 2, 'Numbers should be equal');
✅ Always use strictEqual or assert.strict to avoid coercion bugs.
Equality

assert.notEqual / notStrictEqual

assert.notEqual(actual, expected, message?) / assert.notStrictEqual(actual, expected, message?)

Assert that two values are not equal. Prefer notStrictEqual.

import assert from 'node:assert';

// Loose inequality (avoid)
assert.notEqual(1, '2'); // Passes

// Strict inequality (recommended)
assert.notStrictEqual(1, '1'); // Passes (1 !== '1')
assert.notStrictEqual(1, 2); // Passes
Equality

assert.deepStrictEqual / notDeepStrictEqual

assert.deepStrictEqual(actual, expected, message?) / assert.notDeepStrictEqual(actual, expected, message?)

Deep equality check for objects, arrays, and nested structures. Compares values recursively.

import assert from 'node:assert';

// Deep strict equality
const obj1 = { a: 1, b: { c: 2 } };
const obj2 = { a: 1, b: { c: 2 } };
assert.deepStrictEqual(obj1, obj2); // Passes

// Arrays
const arr1 = [1, [2, 3]];
const arr2 = [1, [2, 3]];
assert.deepStrictEqual(arr1, arr2); // Passes

// Different types fail
assert.deepStrictEqual([1], { 0: 1 }); // Throws

// Not deep strict equal
assert.notDeepStrictEqual({ a: 1 }, { a: 2 }); // Passes
✅ Use deepStrictEqual for comparing objects, arrays, and complex structures.
🎯 Comparison Assertions
Comparison

assert.ok / assert.match

assert.ok(value, message?) / assert.match(string, regexp, message?) / assert.doesNotMatch(string, regexp, message?)

Truthiness and pattern matching assertions.

import assert from 'node:assert';

// assert.ok: check truthiness
assert.ok(true);           // Passes
assert.ok(1);             // Passes
assert.ok('hello');       // Passes
assert.ok(false, 'Must be truthy'); // Throws
assert.ok(0);            // Throws
assert.ok(null);         // Throws

// assert.match: regex match
assert.match('hello world', /world/); // Passes
assert.match('foo bar', /baz/);        // Throws

// assert.doesNotMatch: no regex match
assert.doesNotMatch('hello world', /baz/); // Passes
✅ assert.ok is great for checking existence or truthiness of values.
Comparison

assert.throws / assert.doesNotThrow

assert.throws(fn, error?, message?) / assert.doesNotThrow(fn, message?)

Test if a function throws an error. Can match error type, message, or instance.

import assert from 'node:assert';

// Assert that a function throws
assert.throws(() => {
  throw new Error('Something went wrong');
});

// Match error message
assert.throws(
  () => { throw new Error('Invalid input'); },
  /invalid/i
);

// Match error constructor
assert.throws(
  () => { throw new TypeError('Bad type'); },
  TypeError
);

// Match both
assert.throws(
  () => { throw new RangeError('Out of range'); },
  { name: 'RangeError', message: /range/i }
);

// Assert that a function does NOT throw
assert.doesNotThrow(() => {
  // No error thrown
});
✅ Essential for testing error handling and validation logic.
🚨 Error Assertions
Error

assert.ifError

assert.ifError(value, message?)

Assert that a value is null or undefined. Usually used to check callback errors.

import assert from 'node:assert';

// Callback-style error handling
function readFile(path, callback) {
  // Simulate success
  callback(null, 'file content');
}

readFile('file.txt', (err, data) => {
  assert.ifError(err); // Passes if err is null/undefined
  console.log(data);
});

// With error
function badRead(callback) {
  callback(new Error('Failed'));
}

badRead((err) => {
  assert.ifError(err); // Throws with the error
});
✅ Perfect for testing callback-based APIs and ensuring no errors occurred.
Error

assert.fail

assert.fail(message?)

Force an assertion failure. Useful for marking unreachable code or expected failures.

import assert from 'node:assert';

// Mark code that should never be reached
function processValue(value) {
  switch (value) {
    case 'a':
      return 1;
    case 'b':
      return 2;
    default:
      assert.fail(`Unexpected value: ${value}`);
  }
}

// In tests, to assert a failure case
try {
  processValue('c');
} catch (err) {
  assert.strictEqual(err.message, 'Unexpected value: c');
}
🧪 Utility Assertions
Util

assert.strict / assert.deepEqual (deprecated)

assert.strict (object) / assert.deepEqual (deprecated)

The strict object provides all strict versions of assertions. deepEqual is deprecated — use deepStrictEqual.

import { strict } from 'node:assert';

// All methods in 'strict' are strict by default
strict.equal(1, 1);                 // strictEqual
strict.notEqual(1, '1');             // notStrictEqual
strict.deepEqual({ a: 1 }, { a: 1 }); // deepStrictEqual

// ❌ DEPRECATED: avoid using assert.deepEqual
import assert from 'node:assert';
assert.deepEqual(1, '1'); // Passes (coercion) - avoid!

// ✅ Use deepStrictEqual instead
assert.deepStrictEqual(1, 1);
✅ Always use strict import for modern, coercion-free assertions.
Util

assert.snapshot / assert.rejects

assert.rejects(asyncFn, error?, message?) / assert.doesNotReject(asyncFn, message?)

Test Promise rejection. Async/await friendly alternatives to throws.

import assert from 'node:assert';

// Test async function that rejects
async function asyncError() {
  throw new Error('Async error');
}

// assert.rejects
await assert.rejects(
  async () => await asyncError(),
  /Async error/
);

// Test promise rejection
await assert.rejects(
  Promise.reject(new Error('Promise error')),
  Error
);

// assert.doesNotReject
await assert.doesNotReject(
  async () => await Promise.resolve('ok')
);
✅ Use assert.rejects for testing async functions that should throw.
Util

assert.AssertionError

new assert.AssertionError(options)

Custom assertion error class. Useful for creating custom assertions.

import assert from 'node:assert';

// Create a custom assertion error
const err = new assert.AssertionError({
  message: 'Custom assertion failed',
  actual: 42,
  expected: 43,
  operator: 'strictEqual'
});

// Use in custom assertion
function customAssert(condition, message) {
  if (!condition) {
    throw new assert.AssertionError({
      message: message || 'Custom assertion failed',
      actual: condition,
      expected: true,
      operator: 'customAssert'
    });
  }
}

// Check if error is from assert
try {
  customAssert(false);
} catch (err) {
  if (err instanceof assert.AssertionError) {
    console.log('This is an assertion error');
  }
}
🚀 Pro Pattern — Custom Validation with Assertions: 
import assert from 'node:assert';

// Input validation function
function validateUser(user) {
  assert.strictEqual(typeof user, 'object', 'User must be an object');
  assert.ok(user.name, 'User must have a name');
  assert.match(user.email, /^[^\s@]+@[^\s@]+\.[^\s@]+$/, 'Invalid email format');
  assert.ok(user.age >= 18, 'User must be 18 or older');
  assert.deepStrictEqual(
    Object.keys(user),
    ['name', 'email', 'age'],
    'User object has unexpected properties'
  );
  return user;
}

// Production validation with graceful error handling
function safeValidate(user) {
  try {
    return { valid: true, data: validateUser(user) };
  } catch (err) {
    if (err instanceof assert.AssertionError) {
      return { valid: false, error: err.message };
    }
    throw err;
  }
}

⚠️ Common Pitfalls

MistakeWhy It's WrongFix
Using equal instead of strictEqual Type coercion hides bugs (1 == '1')Use strictEqual or assert.strict
Using deepEqual instead of deepStrictEqual Coerces values in objectsUse deepStrictEqual
Not including error messages Hard to debug assertion failuresAlways add descriptive messages
Using assert.ok with false values 0, '', null, undefined all failBe explicit: assert.strictEqual(val, expected)
Not handling assert.throws async Doesn't work with PromisesUse assert.rejects for async
Assertions in production They throw and crash the processUse try/catch or only in development
