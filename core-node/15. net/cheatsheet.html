🌐 Node.js net Module Cheat Sheet

TCP networking — servers, clients, and sockets — Node.js 18+

💡 Golden Rule: The net module provides TCP (not HTTP) networking. Use net.createServer() for TCP servers, net.connect() for clients. Always handle errors and clean up sockets. Import with node:net prefix.
🖥️ TCP Servers
Server

net.createServer

net.createServer(options?, connectionListener?)

Create a TCP server. Listens for connections and emits 'connection' events with socket objects.

import net from 'node:net';

// Create server with connection listener
const server = net.createServer((socket) => {
  console.log('Client connected:', socket.remoteAddress);

  // Handle data
  socket.on('data', (data) => {
    console.log('Received:', data.toString());
    socket.write('Echo: ' + data);
  });

  // Handle client disconnect
  socket.on('end', () => {
    console.log('Client disconnected');
  });

  // Handle errors
  socket.on('error', (err) => {
    console.error('Socket error:', err);
  });
});

// Start listening
server.listen(3000, () => {
  console.log('Server listening on port 3000');
});
✅ Each connection gets a Socket object. Use socket.write() to send data.
Server

server.listen / server.close

server.listen(port?, host?, backlog?, callback?) / server.close(callback?)

Start and stop the server. listen can bind to port, host, or a Unix socket.

import net from 'node:net';

const server = net.createServer();

// Listen on port 3000
server.listen(3000, () => {
  console.log('Server listening');
});

// Listen on port with host
server.listen(3000, '127.0.0.1');

// Listen on a random port (0 = any available)
server.listen(0, () => {
  console.log('Random port:', server.address().port);
});

// Listen on Unix socket (Linux/Mac)
server.listen('/tmp/socket.sock');

// Close server
server.close(() => {
  console.log('Server closed');
});
✅ Use server.address() to get the bound address after listening.
Server

Server Events & Properties

server.on('connection'), server.maxConnections, server.getConnections()

Monitor server activity and set connection limits.

import net from 'node:net';

const server = net.createServer();

// Connection event
server.on('connection', (socket) => {
  console.log('New connection');
});

// Listening event
server.on('listening', () => {
  console.log('Server started');
});

// Error event
server.on('error', (err) => {
  console.error('Server error:', err);
});

// Close event
server.on('close', () => {
  console.log('Server closed');
});

// Set max connections
server.maxConnections = 100;

// Get active connections
server.getConnections((err, count) => {
  console.log(`Active connections: ${count}`);
});
🔌 TCP Clients
Client

net.connect / net.createConnection

net.connect(port, host?, connectListener?) / net.createConnection(...)

Create a TCP client connection. Returns a Socket object. createConnection is the preferred method.

import net from 'node:net';

// Connect to a server
const client = net.createConnection({ port: 3000, host: '127.0.0.1' }, () => {
  console.log('Connected to server!');
  client.write('Hello from client');
});

// Or with separate arguments
const client2 = net.connect(3000, 'localhost', () => {
  console.log('Connected');
});

// Connect to Unix socket
const client3 = net.createConnection('/tmp/socket.sock');

// Handle data from server
client.on('data', (data) => {
  console.log('Server says:', data.toString());
});

client.on('end', () => {
  console.log('Server closed connection');
});

client.on('error', (err) => {
  console.error('Connection error:', err);
});
✅ createConnection is preferred over connect (they are aliases, but createConnection is more consistent with other Node.js APIs).
🔌 Socket API
Socket

socket.write / socket.end

socket.write(data, encoding?, callback?) / socket.end(data?, encoding?, callback?)

Send data to the connected peer. end sends data and closes the connection.

import net from 'node:net';

const server = net.createServer((socket) => {
  // Write data (String or Buffer)
  socket.write('Hello client\n');
  socket.write(Buffer.from([0x48, 0x69])); // 'Hi'

  // Write with callback
  socket.write('Data sent', 'utf-8', () => {
    console.log('Write complete');
  });

  // End the connection (send FIN packet)
  socket.end('Goodbye\n', () => {
    console.log('Connection ended');
  });
});
⚠️ socket.write() may buffer data internally. Check socket.writableLength and the 'drain' event for backpressure handling.
Socket

socket.pipe / socket.unpipe

socket.pipe(destination, options?) / socket.unpipe(destination?)

Pipe data from a socket to another stream (e.g., another socket, file, or process). Great for proxying.

import net from 'node:net';

// Simple TCP proxy
const proxy = net.createServer((clientSocket) => {
  // Connect to backend
  const backend = net.createConnection({ port: 8080, host: 'backend' });

  // Pipe data both ways
  clientSocket.pipe(backend);
  backend.pipe(clientSocket);

  // Handle errors
  clientSocket.on('error', () => backend.destroy());
  backend.on('error', () => clientSocket.destroy());
});
✅ Piping automatically handles backpressure and is very efficient for streaming data.
Socket

socket.setTimeout / socket.setKeepAlive / socket.setNoDelay

socket.setTimeout(timeout, callback?) / socket.setKeepAlive(enable, initialDelay?) / socket.setNoDelay(noDelay?)

Control socket behavior: timeouts, keep-alive, and Nagle's algorithm.

import net from 'node:net';

const server = net.createServer((socket) => {
  // Set idle timeout (10 seconds)
  socket.setTimeout(10000, () => {
    console.log('Socket timeout');
    socket.end();
  });

  // Enable TCP keep-alive (prevents stale connections)
  socket.setKeepAlive(true, 30000); // 30 seconds initial delay

  // Disable Nagle's algorithm (send data immediately)
  socket.setNoDelay(true);
});

// Reset timeout after receiving data
socket.on('data', () => {
  socket.setTimeout(10000); // Reset the timer
});
✅ Use setNoDelay(true) for low-latency applications. Use setKeepAlive to detect dead connections.
Socket

Socket Properties & Events

socket.remoteAddress, socket.remotePort, socket.localAddress, socket.localPort

Get connection details. Listen for socket events.

import net from 'node:net';

const server = net.createServer((socket) => {
  // Connection info
  console.log(`Remote: ${socket.remoteAddress}:${socket.remotePort}`);
  console.log(`Local: ${socket.localAddress}:${socket.localPort}`);

  // Socket state
  console.log('Bytes written:', socket.bytesWritten);
  console.log('Bytes read:', socket.bytesRead);

  // Events
  socket.on('connect', () => console.log('Connected'));
  socket.on('ready', () => console.log('Ready'));
  socket.on('drain', () => console.log('Drain'));
  socket.on('close', (hadError) => console.log(`Closed: ${hadError}`));
  socket.on('error', (err) => console.error(err));
});
🔧 Utility Functions
Util

net.isIP / net.isIPv4 / net.isIPv6

net.isIP(input) / net.isIPv4(input) / net.isIPv6(input)

Check if a string is a valid IP address.

import net from 'node:net';

net.isIP('192.168.1.1');   // 4
net.isIP('::1');          // 6
net.isIP('invalid');     // 0
net.isIPv4('192.168.1.1'); // true
net.isIPv6('::1');        // true
Util

Socket Buffer & Backpressure

socket.writableLength, socket.writableHighWaterMark, socket.writableCorked

Monitor and manage socket write buffers to handle backpressure.

import net from 'node:net';

const server = net.createServer((socket) => {
  function sendData(chunk) {
    // Check if we can write without buffering
    if (socket.writableLength < socket.writableHighWaterMark) {
      const written = socket.write(chunk);
      console.log('Written:', written);
    } else {
      // Buffer is full, wait for drain
      socket.once('drain', () => sendData(chunk));
    }
  }

  // Cork/uncork for batched writes (performance)
  socket.cork();
  socket.write('chunk 1');
  socket.write('chunk 2');
  socket.uncork(); // Flush all writes at once
});
✅ Use cork()/uncork() for performance when writing multiple chunks. Handle backpressure to avoid memory issues.
🚀 Pro Pattern — TCP Echo Server with Graceful Shutdown: 
import net from 'node:net';

const server = net.createServer(socket => {
  socket.on('data', data => {
    // Echo back with prefix
    socket.write('ECHO: ' + data);
  });
  
  socket.on('error', err => {
    console.error('Socket error:', err);
  });
});

// Graceful shutdown
const shutdown = () => {
  console.log('Shutting down...');
  server.close(() => {
    console.log('Server closed');
    process.exit(0);
  });
  
  // Force exit after 5s if graceful close fails
  setTimeout(() => {
    console.log('Force exit');
    process.exit(1);
  }, 5000).unref();
};

process.on('SIGTERM', shutdown);
process.on('SIGINT', shutdown);

server.listen(3000, () => {
  console.log('Server running on port 3000');
});

⚠️ Common Pitfalls

MistakeWhy It's WrongFix
Not handling error eventsUnhandled errors crash the processAlways add socket.on('error')
Assuming data is one messageTCP is a stream — data can be chunkedImplement your own message framing
Not handling backpressureMemory exhaustion from bufferingCheck writableLength and listen for 'drain'
Leaking socketsResource exhaustionAlways socket.end() or socket.destroy()
Binding to all interfaces (0.0.0.0) in productionSecurity riskBind to specific interfaces or localhost
Using socket.write() after socket.end()Throws errorCheck socket.writable before writing
