📦 Node.js zlib Module Cheat Sheet

Compression and decompression — gzip, deflate, brotli — Node.js 18+

💡 Golden Rule: Use zlib for compressing HTTP responses, files, or any data. Brotli offers better compression than gzip but is slower. Import with node:zlib prefix. Always handle errors and memory limits.
🗜️ Compression (Streams)
Gzip

zlib.createGzip / createGunzip

zlib.createGzip(options?) / zlib.createGunzip(options?)

Compress/decompress using gzip format. Great for HTTP compression and file compression.

import zlib from 'node:zlib';
import { createReadStream, createWriteStream } from 'node:fs';

// Compress a file (gzip)
const gzip = zlib.createGzip();
const source = createReadStream('input.txt');
const destination = createWriteStream('input.txt.gz');

source.pipe(gzip).pipe(destination)
  .on('finish', () => console.log('Compression complete'))
  .on('error', (err) => console.error(err));

// Decompress a gzip file
const gunzip = zlib.createGunzip();
const compressed = createReadStream('input.txt.gz');
const decompressed = createWriteStream('output.txt');

compressed.pipe(gunzip).pipe(decompressed)
  .on('finish', () => console.log('Decompression complete'));
✅ Use gzip for general-purpose compression. Compatible with most tools and browsers.
Deflate

zlib.createDeflate / createInflate

zlib.createDeflate(options?) / zlib.createInflate(options?)

Compress/decompress using deflate format (raw zlib without headers).

import zlib from 'node:zlib';

// Compress with deflate
const deflate = zlib.createDeflate({
  level: 9,              // Compression level (0-9)
  memLevel: 8            // Memory usage (1-9)
});

const input = Buffer.from('Hello world!');
const chunks = [];
deflate.on('data', chunk => chunks.push(chunk));
deflate.on('end', () => {
  const compressed = Buffer.concat(chunks);
  console.log('Compressed size:', compressed.length);
  
  // Decompress
  const inflate = zlib.createInflate();
  const decompressedChunks = [];
  inflate.on('data', chunk => decompressedChunks.push(chunk));
  inflate.on('end', () => {
    const decompressed = Buffer.concat(decompressedChunks);
    console.log('Decompressed:', decompressed.toString());
  });
  inflate.write(compressed);
  inflate.end();
});

deflate.write(input);
deflate.end();
✅ Deflate is the foundation of gzip. Use when you need raw zlib format.
Deflate

zlib.createDeflateRaw / createInflateRaw

zlib.createDeflateRaw() / zlib.createInflateRaw()

Compress/decompress without zlib headers (raw deflate). Used in some protocols.

import zlib from 'node:zlib';

// Raw deflate (no headers)
const deflateRaw = zlib.createDeflateRaw();
const inflateRaw = zlib.createInflateRaw();

const data = Buffer.from('Raw deflate compression');

// Compress
const compressed = zlib.deflateRawSync(data);
console.log('Raw compressed:', compressed.toString('base64'));

// Decompress
const decompressed = zlib.inflateRawSync(compressed);
console.log('Raw decompressed:', decompressed.toString());
⚠️ Raw deflate is less common. Only use if you know the other side expects raw format.
🌀 Brotli Compression
Brotli

zlib.createBrotliCompress / createBrotliDecompress

zlib.createBrotliCompress(options?) / zlib.createBrotliDecompress(options?)

Compress/decompress using Brotli algorithm. Better compression than gzip but slower.

import zlib from 'node:zlib';
import { createReadStream, createWriteStream } from 'node:fs';

// Brotli compress with options
const brotliCompress = zlib.createBrotliCompress({
  params: {
    // Quality: 0-11 (higher = better compression, slower)
    [zlib.constants.BROTLI_PARAM_QUALITY]: 11,
    // LGWIN: 10-24 (larger = better compression, more memory)
    [zlib.constants.BROTLI_PARAM_LGWIN]: 22,
    // Mode: TEXT, FONT, or GENERIC
    [zlib.constants.BROTLI_PARAM_MODE]: zlib.constants.BROTLI_MODE_TEXT
  }
});

const source = createReadStream('large-file.txt');
const dest = createWriteStream('large-file.txt.br');

source.pipe(brotliCompress).pipe(dest)
  .on('finish', () => console.log('Brotli compression complete'));

// Decompress Brotli
const brotliDecompress = zlib.createBrotliDecompress();
const compressed = createReadStream('large-file.txt.br');
const output = createWriteStream('decompressed.txt');

compressed.pipe(brotliDecompress).pipe(output)
  .on('finish', () => console.log('Brotli decompression complete'));
✅ Brotli is best for web content (text, HTML, CSS, JS). Most modern browsers support it.
⚡ Synchronous & Promise APIs
Util

Sync compression / decompression

zlib.gzipSync(data, options?) / zlib.gunzipSync(data, options?) / zlib.deflateSync() / zlib.inflateSync() / zlib.brotliCompressSync() / zlib.brotliDecompressSync()

Synchronous versions of compression/decompression. Simpler but blocks the event loop.

import zlib from 'node:zlib';

// Sync gzip compress
const data = Buffer.from('Hello world!'.repeat(1000));
const compressed = zlib.gzipSync(data, { level: 9 });
console.log(`Compressed: ${compressed.length} bytes (from ${data.length})`);

const decompressed = zlib.gunzipSync(compressed);
console.log(`Decompressed: ${decompressed.length} bytes`);

// Sync deflate
const deflated = zlib.deflateSync(data);
const inflated = zlib.inflateSync(deflated);

// Sync Brotli
const brCompressed = zlib.brotliCompressSync(data, {
  params: {
    [zlib.constants.BROTLI_PARAM_QUALITY]: 11
  }
});
const brDecompressed = zlib.brotliDecompressSync(brCompressed);
⚠️ Sync methods block the event loop. Use streams or promises for large data.
Util

Promise-based compression (Node.js 15+)

zlib.gzip(data, options?) / zlib.gunzip(data, options?) / zlib.brotliCompress(data, options?)

Callback-style functions that can be promisified. Use with util.promisify.

import zlib from 'node:zlib';
import { promisify } from 'node:util';

const gzip = promisify(zlib.gzip);
const gunzip = promisify(zlib.gunzip);
const brotliCompress = promisify(zlib.brotliCompress);
const brotliDecompress = promisify(zlib.brotliDecompress);

// Async/await usage
async function compressData(data) {
  try {
    const compressed = await gzip(data, { level: 9 });
    console.log(`Compressed to ${compressed.length} bytes`);
    
    const decompressed = await gunzip(compressed);
    return { compressed, decompressed };
  } catch (err) {
    console.error('Compression error:', err);
  }
}

// Brotli with quality control
const br = await brotliCompress(Buffer.from('Hello world'), {
  params: {
    [zlib.constants.BROTLI_PARAM_QUALITY]: 11
  }
});
✅ Promisified versions are great for async workflows with moderate data sizes.
🔧 Advanced Options
Util

Compression levels & options

zlib.constants (level, memLevel, strategy, etc.)

Fine-tune compression with options like level, memory, and strategy.

import zlib from 'node:zlib';

// Gzip options
const gzip = zlib.createGzip({
  level: 9,                    // 0-9 (0=no compression, 9=best)
  memLevel: 8,                 // 1-9 (memory usage)
  strategy: zlib.constants.Z_RLE, // Strategy for compression
  windowBits: 15,               // 8-15 (window size)
  chunkSize: 16 * 1024,        // Internal buffer size
  flush: zlib.constants.Z_SYNC_FLUSH
});

// Compression strategies
// Z_DEFAULT_STRATEGY - default
// Z_FILTERED - for data with lots of small values
// Z_HUFFMAN_ONLY - only Huffman coding
// Z_RLE - run-length encoding
// Z_FIXED - fixed Huffman tables

// Brotli options
const brotli = zlib.createBrotliCompress({
  params: {
    [zlib.constants.BROTLI_PARAM_QUALITY]: 11,        // 0-11
    [zlib.constants.BROTLI_PARAM_LGWIN]: 24,          // 10-24
    [zlib.constants.BROTLI_PARAM_MODE]: zlib.constants.BROTLI_MODE_TEXT,
    [zlib.constants.BROTLI_PARAM_SIZE_HINT]: 1024        // Hint for better compression
  }
});
⚠️ Higher compression levels use more CPU and memory. Balance based on your use case.
Util

Memory limits & error handling

zlib.constants.Z_FINISH / zlib.constants.Z_NO_FLUSH / zlib.constants.Z_SYNC_FLUSH

Control flushing behavior and memory limits to prevent out-of-memory errors.

import zlib from 'node:zlib';

// Flush options
const deflate = zlib.createDeflate({
  flush: zlib.constants.Z_SYNC_FLUSH // Flush data immediately
});

// Manual flush
deflate.write(data);
deflate.flush(zlib.constants.Z_SYNC_FLUSH);

// Set memory limits for protection
const gzip = zlib.createGzip({
  memLevel: 4,          // Lower memory usage
  windowBits: 12        // Smaller window (less memory)
});

// Error handling is critical
const gunzip = zlib.createGunzip();
gunzip.on('error', (err) => {
  if (err.code === 'Z_BUF_ERROR') {
    console.error('Buffer error - data may be corrupted');
  } else if (err.code === 'Z_DATA_ERROR') {
    console.error('Data corrupted');
  } else {
    console.error('Compression error:', err);
  }
});
✅ Always handle errors. Corrupted or incomplete data can cause fatal zlib errors.
🌐 HTTP Compression
HTTP

zlib with HTTP (Accept-Encoding)

HTTP compression middleware style

Compress HTTP responses based on client's Accept-Encoding header.

import zlib from 'node:zlib';
import http from 'node:http';

const server = http.createServer((req, res) => {
  const acceptEncoding = req.headers['accept-encoding'] || '';
  const data = Buffer.from('Hello world! '.repeat(1000));

  // Choose encoding based on client support
  if (acceptEncoding.includes('br')) {
    // Brotli (best compression)
    res.writeHead(200, { 'Content-Encoding': 'br' });
    const compress = zlib.createBrotliCompress();
    compress.pipe(res);
    compress.write(data);
    compress.end();
  } else if (acceptEncoding.includes('gzip')) {
    // Gzip (fallback)
    res.writeHead(200, { 'Content-Encoding': 'gzip' });
    const compress = zlib.createGzip();
    compress.pipe(res);
    compress.write(data);
    compress.end();
  } else {
    // No compression
    res.end(data);
  }
});

server.listen(3000, () => {
  console.log('HTTP server with compression');
});
✅ Always check Accept-Encoding and use the best supported algorithm (br > gzip > deflate).
🚀 Pro Pattern — Compress String with Best Algorithm: 
import zlib from 'node:zlib';

// Smart compression based on data size and type
async function compress(data, options = {}) {
  const input = typeof data === 'string' 
    ? Buffer.from(data, 'utf-8') 
    : data;

  // For small data, compression may not help
  if (input.length < < 100) {
    return { compressed: input, algorithm: 'none', size: input.length };
  }

  // Try Brotli first (best compression)
  try {
    const br = await zlib.brotliCompress(input, {
      params: { [zlib.constants.BROTLI_PARAM_QUALITY]: options.quality || 6 }
    });
    return { compressed: br, algorithm: 'brotli', size: br.length };
  } catch (err) {
    // Fallback to gzip
    const gzip = await zlib.gzip(input, { level: options.level || 6 });
    return { compressed: gzip, algorithm: 'gzip', size: gzip.length };
  }
}

// Usage
const result = await compress('Hello world '.repeat(1000));
console.log(`Compressed with ${result.algorithm}: ${result.size} bytes`);

MistakeWhy It's WrongFix
Not handling errorsCorrupted data crashes the processAlways add .on('error') or try/catch
Using sync methods in serversBlocks the event loopUse streams or promises with async/await
Compressing already compressed dataWastes CPU, may increase sizeCheck content type or file extension
Memory limits exceededOut of memory errorsUse streams for large files, adjust memLevel
Assuming all browsers support BrotliOlder browsers don't support itCheck Accept-Encoding header
Not flushing dataData may be buffered indefinitelyUse flush() or let stream end naturally
