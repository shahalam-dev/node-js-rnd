🌐 Node.js http / https Module Cheat Sheet

Build servers and make requests — Node.js 18+

💡 Golden Rule: For new code, prefer the built-in fetch() API (Node 18+) for HTTP requests. Use http/https modules for servers, low-level control, or when you need streaming.
🖥️ Creating a Server
Server

http.createServer

http.createServer(requestListener)

Create an HTTP server. The callback receives (req, res) for every incoming request.

import http from 'node:http';

const server = http.createServer((req, res) => {
  console.log(req.method, req.url);  // GET /users

  res.writeHead(200, { 'Content-Type': 'application/json' });
  res.end(JSON.stringify({ message: 'Hello World' }));
});

server.listen(3000, () => {
  console.log('Server running on http://localhost:3000');
});
Server

HTTPS Server

https.createServer(options, requestListener)

Create an HTTPS server with SSL/TLS certificate.

import https from 'node:https';
import { readFileSync } from 'node:fs';

const options = {
  key: readFileSync('./server-key.pem'),
  cert: readFileSync('./server-cert.pem')
};

https.createServer(options, (req, res) => {
  res.end('Secure connection!');
}).listen(443);
🔒 Use https for production. For local dev, consider mkcert for self-signed certs.
Server

server.listen

server.listen(port, hostname?, callback?)

Start listening for connections. Multiple signatures supported.

// Port only (accepts all interfaces)
server.listen(3000);

// Port + callback
server.listen(3000, () => console.log('Ready'));

// Port + hostname + callback
server.listen(3000, '127.0.0.1', () => {});

// Unix socket (faster, same-machine)
server.listen('/tmp/app.sock');
📨 The Request Object (req)
Req

req Properties

IncomingMessage — read-only stream with request metadata.

req.method        // 'GET', 'POST', 'PUT', 'DELETE', etc.
req.url           // '/users?id=123' (raw URL string)
req.headers       // { 'content-type': 'application/json', ... }
req.httpVersion   // '1.1' or '2.0'

// Parse URL
const url = new URL(req.url, `http://${req.headers.host}`);
url.pathname      // '/users'
url.searchParams.get('id')  // '123'
Req

Reading Request Body

req is a readable stream. Collect chunks to get the full body.

function getBody(req) {
  return new Promise((resolve, reject) => {
    let body = '';
    req.on('data', chunk => body += chunk);
    req.on('end', () => resolve(body));
    req.on('error', reject);
  });
}

// Usage
const body = await getBody(req);
const data = JSON.parse(body);
⚠️ For large uploads, use streams or formidable/multer instead of buffering.
📤 The Response Object (res)
Res

res.writeHead / res.statusCode

Set status code and headers. Must be called before res.end() (unless using implicit 200).

// Set status + headers
res.writeHead(200, {
  'Content-Type': 'application/json',
  'Cache-Control': 'no-cache'
});

// Or set individually
res.statusCode = 404;
res.setHeader('Content-Type', 'text/plain');

// Multiple values for same header
res.setHeader('Set-Cookie', ['a=1', 'b=2']);
Res

res.write / res.end

Send response data. end() finishes the response and optionally sends final data.

// Send and close
res.end('Hello');

// Write multiple chunks, then end
res.write('Chunk 1');
res.write('Chunk 2');
res.end('Final');

// JSON response
res.writeHead(200, { 'Content-Type': 'application/json' });
res.end(JSON.stringify({ ok: true }));

// Redirect
res.writeHead(302, { 'Location': '/new-page' });
res.end();
Res

Streaming Response

Pipe a readable stream directly to the response for efficient file serving.

import fs from 'node:fs';
import path from 'node:path';

server.on('request', (req, res) => {
  const filePath = path.join('./public', req.url);
  const stream = fs.createReadStream(filePath);

  stream.on('open', () => {
    res.writeHead(200, { 'Content-Type': 'text/html' });
  });

  stream.pipe(res);

  stream.on('error', () => {
    res.writeHead(404);
    res.end('Not found');
  });
});
📡 Making HTTP Requests
Request

http.get / http.request

http.get(url/options, callback)

Make HTTP requests. get is shorthand for GET with auto-end. request supports all methods.

import http from 'node:http';

// Simple GET
http.get('http://api.example.com/users', (res) => {
  let data = '';
  res.on('data', chunk => data += chunk);
  res.on('end', () => console.log(JSON.parse(data)));
});

// Full request with POST
const options = {
  hostname: 'api.example.com',
  port: 80,
  path: '/users',
  method: 'POST',
  headers: {
    'Content-Type': 'application/json'
  }
};

const req = http.request(options, (res) => {
  console.log('Status:', res.statusCode);
});

req.write(JSON.stringify({ name: 'John' }));
req.end();
Request

https.request

https.request(options, callback)

Same API as http.request but for HTTPS. Handles TLS automatically.

import https from 'node:https';

https.get('https://api.github.com/users/nodejs', {
  headers: { 'User-Agent': 'node-app' }
}, (res) => {
  let data = '';
  res.on('data', chunk => data += chunk);
  res.on('end', () => console.log(data));
});
🔧 Server Utilities
Util

server.close / server.on

Graceful shutdown and event handling.

// Graceful shutdown
server.close(() => {
  console.log('Server closed');
});

// Event listeners
server.on('connection', (socket) => {
  console.log('New connection from', socket.remoteAddress);
});

server.on('error', (err) => {
  if (err.code === 'EADDRINUSE') {
    console.error('Port already in use');
  }
});

// Get server address
const addr = server.address();
// → { port: 3000, family: 'IPv4', address: '127.0.0.1' }
Util

URL Class (Modern Parsing)

The built-in URL class is the modern way to parse URLs in Node.js.

const url = new URL('https://user:pass@api.example.com:8080/users?id=123#section');

url.protocol     // 'https:'
url.hostname     // 'api.example.com'
url.port         // '8080'
url.pathname     // '/users'
url.search       // '?id=123'
url.hash         // '#section'
url.username     // 'user'
url.password     // 'pass'

// Query params
url.searchParams.get('id')     // '123'
url.searchParams.has('id')     // true
url.searchParams.append('page', '2');

// With req.url (relative)
const parsed = new URL(req.url, `http://${req.headers.host}`);
⚠️ Important: For production APIs, always validate and sanitize req.url to prevent path traversal attacks. Never serve files based directly on user-provided paths without checking against an allowlist.
🚀 Pro Pattern — Minimal Router: 
import http from 'node:http';

const routes = {
  'GET /users': async (req, res) => {
    res.writeHead(200, { 'Content-Type': 'application/json' });
    res.end(JSON.stringify([{ id: 1, name: 'Alice' }]));
  },
  'POST /users': async (req, res) => {
    const body = await getBody(req);
    const user = JSON.parse(body);
    res.writeHead(201);
    res.end(JSON.stringify({ id: 2, ...user }));
  }
};

http.createServer((req, res) => {
  const url = new URL(req.url, `http://${req.headers.host}`);
  const handler = routes[`${req.method} ${url.pathname}`];

  if (handler) {
    handler(req, res).catch(err => {
      res.writeHead(500);
      res.end(err.message);
    });
  } else {
    res.writeHead(404);
    res.end('Not Found');
  }
}).listen(3000);

GoalUse
Create HTTP serverhttp.createServer()
Create HTTPS serverhttps.createServer(options)
Send JSON responseres.writeHead(200, {'Content-Type':'application/json'}) + res.end(JSON.stringify(data))
Read request bodyCollect req.on('data') chunks
Parse URL query paramsnew URL(req.url, base) + url.searchParams
Make GET requesthttp.get() or fetch()
Make POST requesthttp.request() with method: 'POST'
Stream file to clientfs.createReadStream().pipe(res)
